        try:
            primary_crashstore.save_processed(processed_crash)
        except FailureException:
            secondary_crashstore.save_processed(processed_crash)

class BenchmarkingCrashStorage(CrashStorage):

    def save_processed(self, processed_crash):
        start_time = self.start_timer()
        self.wrapped_crashstore.save_processed(processed_crash)
        end_time = self.end_timer()
        logger.debug(
            '%s save_processed %s',
            self.tag,
            end_time - start_time
        )
