def dynamic_load(class_as_string):
    parts = [
        x.strip()
        for x in class_as_string.split('.')
        if x.strip()
    ]
    module = __import__(
        '.'.join(parts[:-1]),
        globals(),
        locals(),
        []
    )
    obj = module
    for name in parts[1:]:
        obj = getattr(obj, name)
    return obj
