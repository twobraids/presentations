(defun match (p s)
  (cond
   ((null p) (null s))  ;match if both are null,
                        ; but not if p is null and s is not null.
   ((atom (car p))
    (and s
         (equal (car p) (car s))
         (match (cdr p) (cdr s))))
   ((and s (eq (caar p) '?))
    (cond ((match (cdr p) (cdr s))
           (set (cadar p) (car s))
           t)
          (t nil)))
   ((eq (caar p) '*)     ;case when  p  starts with * form
    (cond ((and s (match (cdr p) (cdr s)))
           (set (cadar p) (list (car s))) t)
          ((match (cdr p) s)
           (set (cadar p) nil) t)
          ((and s (match p (cdr s)))
           (set (cadar p) (cons (car s) (eval (cadar p)))) t)
          (t nil)))
   ((and s
         (apply (caar p) (list (car s)))
         (match (cdr p) (cdr s)))
    (set (cadar p) (car s)) t)
   (t nil)))
